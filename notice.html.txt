        const db = firebase.firestore();

        let isUserAdmin = false;
        
        const DISCORD_NOTICE_WEBHOOK = "https://discord.com/api/webhooks/1491075955714686977/lkbFUkqu2fipvYSmKA8T_BVLyrw2jqTOCN0WbaSRzThuQne2f_ofGAkkqPZTHJiSem_g";

        async function sendNoticeWebhook(tag, title, content) {
            const payload = {
                content: "🚨 **[@everyone] 새로운 공식 공지사항이 등록되었습니다.**",
                embeds: [{ title: `${tag} ${title}`, description: content.length > 500 ? content.substring(0, 500) + "..." : content, color: 16735744, timestamp: new Date().toISOString(), footer: { text: "CLIX CLAN OFFICIAL NOTICE" } }]
            };
            fetch(DISCORD_NOTICE_WEBHOOK, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }).catch(e => console.error("웹훅 전송 실패 (클라이언트 환경 차단):", e));
        }

        auth.onAuthStateChanged((user) => {
            if (user && user.emailVerified) {
                db.collection("users").doc(user.uid).get().then((doc) => { 
                    isUserAdmin = doc.exists ? (doc.data().isAdmin === true) : false; 
                    updateAdminUI(); 
                    loadNotices(); 
                }).catch((error) => {
                    console.error("권한 확인 실패:", error);
                    isUserAdmin = false;
                    updateAdminUI();
                    loadNotices();
                });
            } else { 
                isUserAdmin = false; 
                updateAdminUI(); 
                loadNotices(); 
            }
        });

        function updateAdminUI() { 
            document.getElementById('admin-editor').style.display = isUserAdmin ? 'block' : 'none'; 
        }

        function loadNotices() {
            db.collection("notices").orderBy("timestamp", "desc").onSnapshot((querySnapshot) => {
                const container = document.getElementById('notice-container'); 
                container.innerHTML = ""; 
                if (querySnapshot.empty) { 
                    container.innerHTML = '<p style="text-align:center; color:#555;">등록된 공지가 없습니다.</p>'; 
                    return; 
                }
                const adminDisplay = isUserAdmin ? 'block' : 'none';
                
                querySnapshot.forEach((doc) => {
                    const data = doc.data(); 
                    const dateStr = data.timestamp ? data.timestamp.toDate().toLocaleDateString('ko-KR') : '방금 전';
                    container.innerHTML += `
                        <div class="notice-card">
                            <div class="notice-header">
                                <div class="notice-title"><span class="notice-tag">${data.tag || '[공지]'}</span>${data.title}</div>
                                <div style="display: flex; align-items: center; gap: 15px;">
                                    <div class="notice-date">${dateStr}</div>
                                    <button class="btn-del" style="display:${adminDisplay}" onclick="deleteNotice('${doc.id}')">삭제</button>
                                </div>
                            </div>
                            <div class="notice-content">${data.content}</div>
                        </div>`;
                });
            }, (error) => {
                console.error("데이터 로드 오류:", error);
                document.getElementById('notice-container').innerHTML = '<p style="text-align:center; color:#ff3333;">데이터를 불러오는 데 실패했습니다. 데이터베이스 규칙을 확인하십시오.</p>';
            });
        }

        async function saveNotice() {
            if (!isUserAdmin) return alert("권한이 없습니다.");
            const t = document.getElementById('title').value.trim(); 
            const g = document.getElementById('tag').value.trim() || '[공지]'; 
            const c = document.getElementById('content').value.trim();
            
            if(!t || !c) return alert("제목과 내용을 모두 입력하십시오.");
            
            try {
                await db.collection("notices").add({ title: t, tag: g, content: c, timestamp: firebase.firestore.FieldValue.serverTimestamp() });
                await sendNoticeWebhook(g, t, c);
                document.getElementById('title').value = ''; 
                document.getElementById('tag').value = ''; 
                document.getElementById('content').value = '';
            } catch (error) { 
                alert("오류: " + error.message); 
            }
        }

        function deleteNotice(docId) { 
            if (isUserAdmin && confirm("이 공지사항을 삭제하시겠습니까?")) {
                db.collection("notices").doc(docId).delete().catch(error => alert("삭제 실패: " + error.message)); 
            }
        }
