    <script>
        const firebaseConfig = {
            apiKey: "AIzaSyAFSAu2aynX6YPiJEatWy9IQgWgUuTx8ds",
            authDomain: "korclix.firebaseapp.com",
            projectId: "korclix",
            storageBucket: "korclix.firebasestorage.app",
            messagingSenderId: "472792477536",
            appId: "1:472792477536:web:14907e228c343b72fd7805"
        };

        firebase.initializeApp(firebaseConfig);
        const auth = firebase.auth();
        const db = firebase.firestore();

        let isUserAdmin = false;

        document.getElementById('match-date').valueAsDate = new Date();

        auth.onAuthStateChanged((user) => {
            if (user && user.emailVerified) {
                db.collection("users").doc(user.uid).get().then((doc) => {
                    isUserAdmin = doc.exists ? (doc.data().isAdmin === true) : false;
                    updateAdminUI();
                    loadScrims();
                });
            } else {
                isUserAdmin = false;
                updateAdminUI();
                loadScrims();
            }
        });

        function updateAdminUI() {
            document.getElementById('admin-editor').style.display = isUserAdmin ? 'block' : 'none';
        }

        function loadScrims() {
            db.collection("scrims").orderBy("date", "desc").onSnapshot((querySnapshot) => {
                const container = document.getElementById('scrim-container');
                container.innerHTML = ""; 

                if (querySnapshot.empty) {
                    container.innerHTML = '<p style="text-align:center; color:#555; padding: 50px 0;">등록된 매치 기록이 없습니다.</p>';
                    return;
                }

                const adminDisplay = isUserAdmin ? 'block' : 'none';

                querySnapshot.forEach((doc) => {
                    const data = doc.data();
                    
                    container.innerHTML += `
                        <div class="scrim-card">
                            <div class="scrim-header">
                                <div class="scrim-title">${data.title}</div>
                                <div class="scrim-date-badge">${data.date}</div>
                            </div>
                            <div class="scrim-result">${data.result}</div>
                            <div class="scrim-details">${data.details}</div>
                            <div class="btn-del" style="display:${adminDisplay}" onclick="deleteScrim('${doc.id}')">DELETE</div>
                        </div>
                    `;
                });
            });
        }

        function saveScrim() {
            if (!isUserAdmin) return alert("권한이 없습니다.");

            const date = document.getElementById('match-date').value;
            const title = document.getElementById('match-title').value.trim();
            const result = document.getElementById('match-result').value.trim();
            const details = document.getElementById('match-details').value.trim();

            if(!date || !title || !result || !details) {
                return alert("모든 항목을 입력하십시오.");
            }

            db.collection("scrims").add({
                date: date,
                title: title,
                result: result,
                details: details,
                timestamp: firebase.firestore.FieldValue.serverTimestamp()
            }).then(() => {
                document.getElementById('match-title').value = '';
                document.getElementById('match-result').value = '';
                document.getElementById('match-details').value = '';
                alert("기록이 성공적으로 등록되었습니다.");
            }).catch((error) => {
                alert("등록 실패: " + error.message);
            });
        }

        function deleteScrim(docId) {
            if (!isUserAdmin) return alert("권한이 없습니다.");
            if(confirm("이 매치 기록을 영구적으로 삭제하시겠습니까?")) {
                db.collection("scrims").doc(docId).delete()
                .catch((error) => alert("삭제 실패: " + error.message));
            }
        }
    </script>
