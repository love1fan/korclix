        const auth = firebase.auth();

        let isUserAdmin = false; 
        let userNick = "";
        let currentReadId = null;

        // 인증 상태 확인
        auth.onAuthStateChanged((user) => {
            const overlay = document.getElementById('lock-overlay');
            if (user && user.emailVerified) {
                // 로그인 완료 (접근 허용)
                overlay.style.display = 'none';
                
                // 유저 정보 가져오기
                db.collection("users").doc(user.uid).get().then((doc) => {
                    if (doc.exists) { 
                        const data = doc.data(); 
                        userNick = data.nickname || user.email.split('@')[0]; 
                        isUserAdmin = data.isAdmin === true; 
                    } else { 
                        userNick = user.email.split('@')[0]; 
                        isUserAdmin = false; 
                    }
                    document.getElementById('nav-user-info').innerText = userNick + "님"; 
                    
                    // 관리자일 경우 글쓰기 버튼 활성화
                    if(isUserAdmin) {
                        document.getElementById('btn-write-open').style.display = 'block';
                    }

                    // 게시글 로드
                    loadNotices();
                });
            } else {
                // 미로그인 시 차단 화면 표시
                overlay.style.display = 'flex';
            }
        });

        // 공지사항 불러오기
        function loadNotices() {
            const boardBody = document.getElementById('board-body');
            const totalCountSpan = document.getElementById('total-count');

            db.collection("notices").orderBy("isPinned", "desc").orderBy("createdAt", "desc")
            .onSnapshot((snapshot) => {
                if (snapshot.empty) {
                    boardBody.innerHTML = '<div class="empty-board">등록된 공지사항이 없습니다.</div>';
                    totalCountSpan.innerText = "0";
                    return;
                }

                boardBody.innerHTML = "";
                totalCountSpan.innerText = snapshot.docs.length;
                let listNum = snapshot.docs.length;

                snapshot.forEach((doc) => {
                    const data = doc.data();
                    const docId = doc.id;
                    const dateObj = data.createdAt ? data.createdAt.toDate() : new Date();
                    const dateStr = `${dateObj.getFullYear()}.${String(dateObj.getMonth() + 1).padStart(2, '0')}.${String(dateObj.getDate()).padStart(2, '0')}`;
                    
                    const rowClass = data.isPinned ? "board-row notice-pin" : "board-row";
                    const displayNum = data.isPinned ? "필독" : listNum--;

                    const rowHtml = `
                        <div class="${rowClass}" onclick="readNotice('${docId}', '${data.title.replace(/'/g, "\\'")}', '${data.author}', '${dateStr}', \`${data.content.replace(/`/g, "\\`")}\`)">
                            <div class="col-id">${displayNum}</div>
                            <div class="col-title">${data.title}</div>
                            <div class="col-author">${data.author}</div>
                            <div class="col-date">${dateStr}</div>
                        </div>
                    `;
                    boardBody.insertAdjacentHTML('beforeend', rowHtml);
                });
            }, (error) => {
                console.error("공지사항 로드 에러:", error);
                boardBody.innerHTML = '<div class="empty-board">데이터를 불러오는 중 오류가 발생했습니다.</div>';
            });
        }

        // 모달 열기/닫기 공통 함수
        function closeModal(modalId) { document.getElementById(modalId).style.display = 'none'; }
        
        // 공지사항 읽기
        function readNotice(id, title, author, date, content) {
            currentReadId = id;
            document.getElementById('read-title').innerText = title;
            document.getElementById('read-author').innerText = "작성자: " + author;
            document.getElementById('read-date').innerText = "작성일: " + date;
            document.getElementById('read-body').innerText = content;
            
            // 관리자면 삭제 버튼 보이기
            document.getElementById('read-admin-controls').style.display = isUserAdmin ? 'block' : 'none';
            
            document.getElementById('read-modal').style.display = 'flex';
        }

        // 글쓰기 창 열기
        function openWriteModal() {
            document.getElementById('write-title').value = '';
            document.getElementById('write-content').value = '';
            document.getElementById('write-pin').checked = false;
            document.getElementById('write-modal').style.display = 'flex';
        }

        // 공지사항 등록
        function submitNotice() {
            const title = document.getElementById('write-title').value.trim();
            const content = document.getElementById('write-content').value.trim();
            const isPinned = document.getElementById('write-pin').checked;

            if (!title || !content) return alert("제목과 내용을 모두 입력해주세요.");

            db.collection("notices").add({
                title: title,
                content: content,
                author: userNick,
                isPinned: isPinned,
                createdAt: firebase.firestore.FieldValue.serverTimestamp()
            }).then(() => {
                alert("공지사항이 등록되었습니다.");
                closeModal('write-modal');
            }).catch((error) => {
                alert("등록 실패: " + error.message);
            });
        }

        // 공지사항 삭제 (관리자용)
        function deleteNotice() {
            if(!currentReadId) return;
            if(confirm("정말 이 공지사항을 삭제하시겠습니까?")) {
                db.collection("notices").doc(currentReadId).delete().then(() => {
                    alert("삭제되었습니다.");
                    closeModal('read-modal');
                }).catch((error) => {
                    alert("삭제 실패: " + error.message);
                });
            }
        }

        // 로그아웃
        function logout() {
            if(confirm("로그아웃 하시겠습니까?")) {
                auth.signOut().then(() => {
                    location.href = 'index.html';
                });
            }
        }
